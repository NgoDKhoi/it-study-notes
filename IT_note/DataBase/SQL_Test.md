SELECT MaPhim, TenPhim
FROM PHIM
JOIN SUATCHIEU ON PHIM.MaPhim = SUATCHIEU.MaPhim
WHERE SUATCHIEU.NgayChieu = '2025-11-12'

--Cau 2
SELECT MaPhim, TenPhim, COUNT(SUATCHIEU.MaSuat) AS SoLanChieu
FROM PHIM
JOIN SUATCHIEU ON PHIM.MaPhim = SUATCHIEU.MaPhim
WHERE YEAR(SUATCHIEU.NgayChieu) = 2025
GROUP BY PHIM.MaPhim, PHIM.TenPhim

--Cau 3
SELECT P.MaPhim, P.TenPhim, SC.MaSuat, SC.NgayChieu, SC.GioChieu, SUM(TT.DanhGia) AS TongDanhGia
FROM SUATCHIEU AS SC
JOIN PHIM AS P ON SC.MaPhim = P.MaPhim
JOIN THONGTIN AS TT ON SC.MaSuat = TT.MaSuat
GROUP BY P.MaPhim, P.TenPhim, SC.MaSuat, SC.NgayChieu, SC.GioChieu

--Cau 4
SELECT COUNT(THONGTIN.MaKH) AS SoLanXem
FROM KHACHHANG
JOIN THONGTIN ON KHACHHANG.MaKH = THONGTIN.MaKH
WHERE KHACHHANG.HoTenKH = 'Nguyễn Văn A'

--Cau 5
SELECT MaPhim, TenPhim, SUM(TT.GiaVe * TT.SLVe) AS DoanhThu
FROM PHIM
JOIN SUATCHIEU ON Phim.MaPhim = SUATCHIEU.MaPhim
JOIN THONGTIN ON SUATCHIEU.MaSUat = THONGTIN.MaSuat
GROUP BY PHIM.MaPhim, PHIM.TenPHim
